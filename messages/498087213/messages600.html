#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <time.h>
#include <random>
#include <stdio.h>
using namespace std;

const int max_x = 10, max_y = 10;
void out(int massiv[max_x][max_y]); // вывод массива
void gen(int massiv[max_x][max_y], int a, int b); // генерация массива
void slogenie(int massiv[max_x][max_y]);


int main() {
int massiv[max_x][max_y];

gen(massiv, -25, 25);
printf("Generated massiv:\n");
out(massiv);


slogenie(massiv);
printf("\nResult massiv:\n");
out(massiv);
}


void gen(int massiv[max_x][max_y], int a, int b) {
int i, j;
srand(time(NULL) * 1000);// устанавливает начальную точку генерации случайных чисел
for (i = 0; i < max_x; i++)
for (j = 0; j < max_y; j++)
massiv[i][j] = (rand() % (b - a) + a);
}


void out(int massiv[max_x][max_y])
{
int i, j;
for (i = 0; i < max_x; i++) {
for (j = 0; j < max_y; j++)
printf("%4d", massiv[i][j]);
printf("\n");
}
}


void slogenie(int massiv[max_x][max_y]){
int i, j, sum;
for (i=0; i < max_x; i++){
sum = 0;
for (j=0; j<max_y; j++)
sum += massiv[i][j];
massiv[i][max_y-1] = sum;
}

}
