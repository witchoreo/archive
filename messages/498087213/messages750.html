#include <time.h>
#include <random>
#include <stdio.h>
using namespace std;

const int m = 10, n = 10;
void gen(int matrix[m][n], int start, int end); // генерация массива
int min_el(int matrix[m][n]); // Максимальный элемент главной диагонали двумерного целочисленного массива размерностью mxn
void change(int matrix[m][n]); // замена элементов массива выше (на 1) и ниже (на 0) главной диагонали
void out(int matrix[m][n]); // вывод массива


int main(){
int matrix[m][n];
gen(matrix, -100, 100);
printf("Generated matrix:\n");
out(matrix);

change(matrix);
printf("\nChanged matrix:\n");
out(matrix);

cout « "\nMin. element on diagonal: " « min_el(matrix) « endl;
}


void gen(int matrix[m][n], int start, int end){
srand(time(NULL)*1000);
for (int i=0; i<m; i++)
for (int j=0; j<n; j++)
matrix[i][j]=(rand()%(end-start)+start);
}

int min_el(int matrix[m][n]){
int min_e = matrix[0][0];
for (int i=0; i<m; i++)
for (int j = 0; j < n; j++)
if((i==j)&&(matrix[i][j] < min_e))
min_e = matrix[i][j];
return min_e;


}

void change(int matrix[m][n]) {
for (int i=0; i<m; i++)
for (int j=0; j<n; j++) {
if (i>j) matrix[i][j]=0.0;
if (i<j) matrix[i][j]=1.1;
}
}

void out(int matrix[m][n])
{
int i,j;
for (i=0; i<m; i++) {
for (j=0; j<n; j++)
printf("%4d",matrix[i][j]);
printf("\n");
}
}
